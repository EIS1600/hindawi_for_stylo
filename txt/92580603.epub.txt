[1mألعاب الصِّبيان عند العرب [0m

[1m(١) الأَسَنُ [0m

لعبة للصبيان يسمونها: الضَّبْطَة والمَسَّة. 

[1m(٢) الأُل١ [0m

قال امرؤ القيس: 

لِمَنْ زُحْلُوقَةٌ زُلُّ 

بها العَيْنَان تَنْهَلُّ 

ينادي الآخرَ الأُلُّ 

ألا حُلُّوا ألا حُلُّوا 

قال المفضَّلُ في قولِ امرِئ القيس: ألا حُلُّوا. قال: هذا معنى لعبة للصبيان يجتمعون فيأخذون خشبة، فيضعونها على قَوْزٍ من رمل ثم 
يجلس على أحد طرفيها جماعة، وعلى الآخر جماعة، فأي الجماعتين كانت أرزن ارتفعت الأخرى، فينادون أصحاب الطرف الآخر: ألا حُلُّوا: أي 
خفِّفوا عن عددكم حتى نساويكم في التعديل. 

قال: وهذه التي تسميها العرب: الدَّوْدَاة والزُّحلُوقة. قال: تسمى أرجوحة الحَضَرِ: المُطَوِّحَة. 

[1m(٣) البُحَّيْثَى [0m

مثال خُلَّيْطَى، لعبة يلعبون بها بالتراب كالبُحْثَة. 

[1m(٤) البُقَّيْرَى [0m

مثال السُّمَّيْهى لعبة الصبيان، وهي كَوْمَةٌ من تراب وحولها خطوط. وبَقَّر الصبيان: لعبوا البُقَّيْرَى، يأتون إلى موضع قد خُبئ لهم 
فيه شيء فيضربون بأيديهم بلا حفر يطلبونه. 

[1m(٥) البُكْسَة [0m

خرقة يُدَوِّرها الصبيان، ثم يأخذون حجرًا فيدوِّرونه كأنه كرة، ثم يتقامرون بهما، وتُسمى هذه اللُّعبَة: الكُجَّة. 

[1m(٦) البُوصاء [0m

لعبة يلعب بها الصبيان، يأخذون عودًا في رأسه نار، فيديرونه على رءوسهم. 

[1m(٧) بيض [0m

لُعبة للعرب، يقولون: أبيضِي حَبَالًا، وأسيدي حَبَالًا. 

أبْيَضَت المرأة وأباضَت ولدت البيض وكذلك الرجُلُ، وأسادَ الرجُلُ وأسوَدَ وُلِدَ له أسود. 

[1m(٨) الجُبَّاع [0m

سهم صغير يلعب به الصبيان، يجعلون على رأسه تمرة لئلا يعقِر.٢ 

[1m(٩) الحَدَبْدَبَى [0m

لُعبة للنبيط، قال الشيخ ابن بري: وجدت حاشية مكتوبة ليست من أصل الكتاب، وهي: حَدبدَبَى اسم لعبة، وأنشد لسالم بن دارة يهجو مُرَّة بن 
رافع الفزاريِّ: 

حَدَبْدَبى حَدَبْدَبى ياصبيانْ 

إن بني فزارة بن ذبيانْ 

قد طَرَّقَت ناقتهم بإنسانْ 

مُشَيَّأٍ٣ أعجب بخلق الرحْمَنْ 

غلبتُم الناسَ بأكل الجُردان 

وسَرَقِ الجار ونَيْك البُعْرَانْ 

[1m(١٠) الجِعِرَّى [0m

لصبيان الأعراب لُعبة يقال لها: الجِعِرَّى، وذلك بأن يُحمل الصبي بين اثنين على أيديهما (وهي المعروفة عند العامة بمقعد السلطان). 

[1m(١١) جَبَّي جُعَل [0m

قالت الأعراب: لنا لعبة يلعب بها الصبيان نسميها: جَبَّي جُعَل، يضع الصبي رأسه على الأرض ثم ينقلب على الظهر. 

[1m(١٢) الجُمَّاح [0m

الجُمَّاح تمرةٌ تُجعل على رأس خشبة يلعب بها الصبيان، وقيل: الجُمَّاح سهم صغير بلا نَصل مدوَّر الرأس يتعلم به الصبيان الرمي، وقيل: 
بل يلعب به الصبيان يجعلون على رأسه تمرة أو طينًا لئلَّا يعقر، يُرمى به الطائر فيلقيه ولا يقتله حتى يأخذه راميه. 

وقال أبو حنيفة: الجُمَّاح سهم الصبي يُجعل في طرفه تمرًا معلوكًا بقدر عفاص٤ القارورة ليكون أهدى له، أملس وليس له ريش، وربما لم يكن 
له أيضًا فُوقٌ. 

[1m(١٣) الجُنَابَاء [0m

والجُنَابَى لُعْبة للصبيان يتجانب الغلامان، فيعتصم كل واحد من الآخر. 

[1m(١٤) الحَجُّورَة [0m

لُعبة يلعب بها الصبيان يخطون خطًّا مستديرًا، ويقف فيه صبي وهنالك الصبيان معه. 

[1m(١٥) الحَرَز [0m

الحَرَز: الخَطَر. وهو الجوز المحكوك يلعب به الصبي. 

[1m(١٦) الحُزُقَّة [0m

في حديث الشعبيِّ: اجتمع جوارٍ، فأَرِنَّ وأَشِرْنَ، ولَعِبْن الحُزُقَّة، قيل: هي لعبة من اللعب، أُخذت من التحزُّق وهو التجمع. 

[1m(١٧) الحِكَّة [0m

في حديث ابن عمر: أنه مرَّ بغلمان يلعبون بالحِكَّة، فأمر بها فدُفِنَت، هي لُعْبَة لهم يأخذون عظمًا فيحكونه حتى يبيَض، ثم يرمونه 
بعيدًا فمن أخذه فهو الغالب. 

[1m(١٨) الخُذْرُوف [0m

وهو الذي يُسمى: الخَرَّارَة، شيء يدوِّره الصبي بخيط في يده فيُسمع له دوي، وقيل: الخُذْرُوف: عود أو قصبة مشقوقة يُفْرَض في وسطه، ثم 
يُشد بخيط فإذا أُمِرَّ دار وسُمع له حفيف، يلعب به الصبيان (وهي المعروفة بالنعَّار). 

[1m(١٩) خَرَاج [0m

وخَرِيج وتِخْريج كلُّه لُعْبَة لفتيان العرب، وهو أن يمسك أحدهم شيئًا بيده، ويقول لسائرهم: أَخرِجُوا ما في يدي. 

[1m(٢٠) المِخْرَاق [0m

منديل أو نحوه، يُلْوَى فيُضرب به، أو يُلف فَيُفَزَّع به، وهو لعبة يلعب بها الصبيان. قال: 

أجالدهم يوم الحديقة حاسرًا 

كأنَّ يدي بالسيف مِخْراقَ لاعب 

ج: مخاريق. 

[1m(٢١) التَّدْبيح [0m

تدبيح الصبيان، إذا لعبوا، وهو أن يُطَأمِنَ أحدهم ظهره، ليجيء الآخر يعدو من بعيد حتى يركبه، والتدبيح التَّطَأطؤ. 

[1m(٢٢) الدَّبُّوق [0m

لعبة يلعب بها الصبيان (معروفة). 

[1m(٢٣) المِدْحَاة [0m

في حديث أبي رافع: كنت أُلاعب الحسن والحسين — رضوان الله عليهما — بالمَدَاحي، وهي أحجار أمثال القِرَصَة. كانوا يحفرون حُفرة ويدحون 
فيها بتلك الأحجار، فإن وقع الحجر فيها غَلَب صاحبها وإن لم يقع غُلب. 

والدِّحْو هو رمي اللاعب بالحجر والجوز وغيره؛ والمِدْحَاة خشبة يدحى بها الصبي فتمر على وجه الأرض لا تأتي على شيء إلا اجْتَحَفَتْه. 

وقد وصف الأسدي المِدحَاة بقوله: هي المَدَاحي والمسادِي، وهي أحجار أمثال القِرَصَة، وقد حفروا حفرة بقدر ذلك الحجر فيتنحون قليلًا ثم 
يدحون بتلك الأحجار إلى تلك الحفرة، فإن وقع فيها الحجر فقد قَمَر، وإلا فقد قُمرَ، قال: وهو يدحو ويسدو إذا دحاها على الأرض إلى 
الحفرة، والحفرة هي أدْحِيَة. 

[1m(٢٤) الدِّرَكْلَة [0m

الدِّرَكْلَة: لُعْبَة يلعب بها الصبيان، وقيل: هي لعبة للعجم، معرب، قال ابن دريد: أحسبها حبشيَّة معربة، وقال أبو عمرو: هو ضرب من 
الرَّقص. قال الأزهري: قرأت بخط شَمِر، قال: قرئ على أبي عبيد وأنا شاهد في حديث النبي ﷺ أنه مرَّ على أصحاب الدركلة، فقال: جدُّوا يا 
بني أرْفَدَة حتى يعلم اليهود والنصارى أن في ديننا فسحة. 

وجاء: الدِّرَقْلَة، لعبة للعجم، معربة. 

[1m(٢٥) الدُّسَّة [0m

لعبة لصبيان الأعراب. 

[1m(٢٦) الدَّعْلَجَة [0m

لُعْبَة للصبيان يختلفون فيها الجَيْئَة والذَّهَاب. 

وقد دَعْلَجَ الصبيان ودَعْلَجَ الجُرَذ كذلك يُقال إن الصبي لَيُدَعْلِج دعْلَجَة الجُرَذ، يجيء ويذهب. 

[1m(٢٧) الدَّكْر [0m

لُعْبَة يلعب بها الزِّنْج والحَبَش. 

[1m(٢٨) الدُّمَّة [0m

لُعْبَة. 

[1m(٢٩) الدُّوَّامَة [0m

فَلْكَة يرميها الصبي بخيط فتُدَوِّم على الأرض، أي تدور. ودُوَّامة الغلام وهي التي تلعب بها الصبيان تُلَفُّ بسير أو خيط ثم تُرمى 
على الأرض فتدور (وهي المعروفة بالنحلة). 

[1m(٣٠) الأُرْجُوحَة [0m

والمَرْجُوحة التي يلعب بها، وهي خشبة تؤخذ فيوضع وسطها على تلٍّ، ثم يجلس غلام على أحد طرفيها وغلام آخر على الطرف الآخر، فَتَرَجَّح 
الخشبة بهما، ويتحركان فيميل أحدهما بصاحبه الآخر. 

ويُقال للحبل الذي يرتج به: الرُّجَّاحة، والنُّوَّاعة، والنُّوَّاطَة، والطُّوَّاحة. 

[1m(٣١) اليَرْمَع [0m

الخَرَّارة التي تلعب بها الصبيان إذا أديرت سمعت لها صوتًا، وهي الخُذْروف. 

[1m(٣٢) الزَّدْو [0m

كالسَّدو، وهو من لُعَب الصبيان بالجوز والِمزْدَاة موضع ذلك، والغالب عليه الزاي، يسدونه في الحفيرة، وزدا الصبي الجوزَ وبالجوز 
يَزْدُو زدوًا؛ أي: لعب ورمى به في الحفيرة، وتلك الحفيرة هي المِزْدَاة. يقال: أَبْعد المَدَى وازْدُهْ. 

[1m(٣٣) السُّدَّر [0m

اللعبة التي تسمى الطُّبْنُ، وهو خط مستدير، تلعب بها الصبيان. قال ابن الأثير: هو لُعبة يلعب بها يُقامر بها، وتُكسر سينها وتُضم وهي 
فارسية معربة. 

[1m(٣٤) سَفْد اللِّقَاح [0m

لعبة يقال لها سَفْد اللِّقَاح، وذلك انتظام الصبيان بعضهم في إثر بعض كل واحد آخذ بحُجْزَة صاحبه من خلفه. 

[1m(٣٥) الأُسْلُوبة [0m

لعبة للأعراب أو فَعْلَة يفعلونها بينهم، حكاها اللِّحْيَاني وقال: بينهم أسْلُوبَة. 

[1m(٣٦) الشَّعَارير [0m

لُعبة للصبيان، لا يُفرد؛ يقال: لعبنا الشَّعَارير، وهذا لعب الشَّعَارير. 

[1m(٣٧) الشَّفَلَّقَة [0m

لُعبة للحاضرة، وهو أن يَكْسَع الإنسان من خلفه فيَصْرَعَه، وهو الأَسْنُ عند العرب، قال: ويُقال سَانَاه، إذا لعب معه الشَّفَلَّقة. 

[1m(٣٨) الضَّبْطَة [0m

لُعبة للأعراب تُسمى: الضَّبْطَة والمسَّة، وهي الطَّريدَة. 

[1m(٣٩) الطَّبْن [0m

والطَّبْن والطِّبْن خط مستدير، يلعب بها الصبيان، يسمونه الرَّحى، وهذه اللعبة تُسمى السُّدَّر، وأنشد: 

يَبِتْنَ يلعبن حَوالَيَّ الطَّبَن٥ 

الجوهري: الطبنة لعبة يُقال لها بالفارسيَّة: سِدَرة، والجمع طُبَن. 

[1m(٤٠) الطَّثُّ [0m

لَعِب الصبيان، يرمون بخشبة مستديرة عريضة يُدَقَّق أحد رأسيها نحو القُلَة يرمون بها، واسم تلك الخشبة المِطَثَّة. ابن الأعرابي: 
المِطَثَّة القُلَة والمطَثُّ اللَّعب بها. 

والصواب: الطَّثُّ: اللعب بها. 

[1m(٤١) المِطَخَّة [0m

خشبة يُحَدَّد أحد طرفيها، ويلعب بها الصبيان. 

[1m(٤٢) الطَّرِيدَة [0m

لعبة لصبيان الأعراب يُقال لها المَأسَة والمَسَّة. وأطْرَد المُسَابِق صاحبه، قال له: إن سبقتني فلك عليَّ كذا، والإطْرَاد أن تقول: 
إن سبقتني فلك عليَّ كذا، وإن سبقتك فلي عليك كذا. 

[1m(٤٣) عَظْم وَضَّاح٦ [0m

لُعْبة لهم؛ يطرحون بالليل قطعة عظم فمن أصابه فقد غَلَب أصحابه، فيقولون: 

عُظَيْمُ وَضَّاحٍ ضِحَنَّ الليله 

لا تَضِحَنَّ بعدها من لَيْلهْ 

وفي الحديث: «بينا هو يلعب مع الصبيان وهو صغير بعَظْم وَضَّاح مرَّ عليه يهوديٌّ فقال له: لَتَقْتُلَنَّ صَنَاديد هذه القرية.» هي 
اللعبة المذكورة. وكانوا إذا أصابه واحد منهم غلب أصحابه، وكانوا إذا غلب واحد من الفريقين ركب أصحابُه الفريقَ الآخر من الموضع الذي 
يجدونه فيه إلى الموضع الذي رَمَوا به منه. 

[1m(٤٤) العُقَّة [0m

التي يلعب بها الصبيان. 

[1m(٤٥) عَيَافٌ والطَّرِيدَة [0m

لُعْبَتَان لصبيان الأعراب، وقد ذكر الطِّرِمَّاح جواري شبَبْن عن هذه اللعب، فقال: 

قَضَتْ من عَيَافٍ والطَّريدَة حاجَةً 

فهنَّ إلى لهو الحديث خُضُوع 

[1m(٤٦) الفُرَّيْرَة [0m

مصغَّرة مشدَّدة، ما يلعب به الصبيان (عن التاج).٧ 

[1m(٤٧) الفَنْزَجَة [0m

الفَنْزَجَة هو اللعب الذي يُقال له الدَّسْتَبَنْد، يُعنى به رقص المجوس، وفي الصحاح: رقص العجم إذا أخذ بعضهم يد بعض وهم يرقصون، 
وأنشد قول العجاج: 

عكف النبِيط يلعبون الفنزجا 

قال ابن السكيت: هي لعبة لهم تُسمى پنجكان بالفارسيَّة. وفي الصحاح: هو بالفارسيَّة بنجه، ابن الأعرابي: الفنزج لعب النبيط إذا بطروا. 

[1m(٤٨) المُفَايَلَة [0m

والفِيال، لعبة لفتيان الأعراب بالتراب، يخبِّئون الشيء في التراب، ثم يقسمونه قسمين، ثم يقول الخابئ لصاحبه: في أي القسمين هو؟ فإذا 
أخطأ قال له فَالَ رأيُك. 

وقيل يُقال لهذه اللعبة: الطُّبَن والسُّدَّر. 

[1m(٤٩) المِقَثَّة [0m

والمِطَثَّة، لغتان، خشبيَّة مستديرة عريضة يلعب بها الصبيان، ينصبون شيئًا ثم يجتثونه بها عن موضعه، قال ابن دريد: هي شبيهة 
بالخرَّارة. تقول: قثثناه وطثثناه، قثًّا وطثًّا. 

[1m(٥٠) القرْق [0m

الذي يُلعب به، والقرق: لعب السُّدَّر. 

[1m(٥١) القُزَة [0m

لعبة للصبيان تسمى في الحضر: يا مهلهلْه هَالِمَهْ. 

[1m(٥٢) بنت مُقَضِّمَة [0m

هي لعبة تُتخذ من جلود بيض، ويقال لها: بنت قُضَّامة. وقال ابن بري: ولعبة أهل المدينة اسمها بنت قضامة. 

[1m(٥٣) القُفَّيْزَى [0m

من لعب صبيان الأعراب، ينصبون خشبة ثم يتقافزون عليها (لعبة الحصان). 

[1m(٥٤) القُلَة [0m

والمِقْلَى والمِقْلَاء كلها عودان يلعب بهما الصبيان، فالمِقلى العود الكبير الذي يُضرب به، والقُلَة الخشبة الصغيرة التي تنصب وهي 
قدر ذراع، والقالي: الذي يلعب فيضرب القلة بالمقلى. والجمع قُلات وقلون وقُلون. 

[1m(٥٥) قَلَوبَع [0m

لُعْبَة. 

[1m(٥٦) القِنِّين [0m

لعبة للروم يتقامرون بها، والتقنين الضرب بالقنين، وهو الطنبور بالحبشيَّة. 

[1m(٥٧) الكُبْنَة [0m

لعبة للأعراب، تُجمع على: كُبَن، وأنشد: 

تدَكَّلت بعدي وألهتها الكُبَن 

ونحن نعدو في الخبار والجرن 

تدكلت، أي: تدللت. 

[1m(٥٨) الكُجَّة [0m

لعبة للصبيان، وهو أن يأخذ الصبي خرْقة فيدورها ويجعلها كأنها كرة، ثم يتقامرون بها، وكجَّ الصبي لعب الكُجَّة، وتُسمى هذه اللعبة 
باسمين، والخرقة يقال لها: التون. والأجرة يقال: لها البُكسة. 

[1m(٥٩) الكُرْك [0m

الكرك والكرج: الذي يُلعب به. 

[1m(٦٠) المَسَّة [0m

الأَسْنُ لُعْبَة لهم يسمونها المَسَّة والضَّبَطة والطَّريدة، لعبة تسميها العامة المَسَّة والضَّبَطَة؛ فإذا وقعت يد الَّلاعب من 
الرجل على بدنه رأسه أو كتفه، فهي: المَسَّة فإذا وقعت على رجله، فهي: الأَسْن. 

[1m(٦١) الأنْبُوثَة [0m

لعبة يلعب بها الصبيان، يحفرون حَفِيرًا ويدفنون فيه شيئًا، فمن استخرجه فقد غلب. 

[1m(٦٢) المنْجَار [0m

لعبة للصبيان يلعبون بها، قال: 

والورد يسعى بِعُصْم في رِحالهم 

كأنَّه لاعب يسعى بمِنْجَار 

[1m(٦٣) الهَبهاب [0m

لُعبة لصبيان العراق. 

[1m(٦٤) المِهْزَام [0m

لعبة لهم يلعبونها يُغطَّى رأس أحدهم ثم يُلطم، وفي رواية: ثم تُضرب استه، ويقال له: من لطمك؟ قال ابن الأثير: وهي: العميضا. 

وقيل: المهزام عود يُجعل في رأسه نار، تلعب به صبيان الأعراب. 

[1m(٦٥) وَضَّاح٨ [0m

في حديث المبعث، أن النبي ﷺ كان يلعب وهو صغير مع الغلمان بعَظْم وَضَّاح، وهي لعبة لصبيان الأعراب يعمدون إلى عظم أبيض فيرمونه في 
ظلمة الليل ثم يتفرقون في طلبه، فمن وجده منهم فله القَمْر، قال: ورأيت الصبيان يصغرونه فيقولون: عُظَيْم وَضَّاح، وأنشد بعضهم: 

عُظَيْم وَضَّاح ضِحَنَّ الليله 

لا تَضِحَنَّ بعدها من لَيْلَه 

قوله: ضحنَّ، أمر من وضح يضح، وتثقيل النون المؤكدة. ومعناه: اظهرن، ووضاح من الوضوح الظهور. 

١ يسبق إلى ذهن القارئ من جعل كلمة (الأُلِّ) عنوانًا لهذه اللعبة أنها تسمى كذلك، كدأب الكاتب في جعل أسماء اللعب الأخرى عنوانات لها؛ 
والحق أن تلك الكلمة ليست اسمًا وإنما هي كلمة وردت في وصف امرئ القيس للعبة. وهي لغة في الأوَّل. أما اسم اللعبة فهو (الدوداة) أو 
(الزحلوقة) كما جاء بعد ذلك في المقال. قال الشاعر: «كأنني فوق دوداة تقلبني.» (انظر لسان العرب: ج١٣ مادة (أل ل) وج٤ مادة دود) 
(التحرير). 

٢ هذا التعريف أورده صاحب اللسان نقلًا عن كراع، ثم قال: «قال ابن سيده: ولا أحقها، وإنما هو الجماح والجماع.» ا.ﻫ. ومن هذا يمكن 
الاستغناء عن اللعبة الثانية عشرة وهي الجُمَّاح. (التحرير). 

٣ المشيَّأ: مختلف الخلق، كأن فيه من كل قبح شيئًا. 

٤ العفاص (بكسر العين) صمام القارورة. 

٥ الطَّبَنُ هنا مصدر؛ لأنه ضرب من اللَّعب. 

٦ انظر الكلمة المرقومة ٦٥. 

٧ كونها مصغرة مشددة يقتضي أن يكون ضبطها فُرَّيَرة. (التحرير). 

٨ ارجع إلى الكلمة المرقومة ٤٣.
